//bubble sort
#include<iostream>
#include<cmath>
#include<cstdlib>
using namespace std;
//5 4 3 2 1
int main(){
int a[5]={},i;
for(int k=1;k<=5;k++) cin>>a[k];
for(i=1;i<=6;i++){
	while((a[i]<a[i-1])){
	int temp;  //4<5
	temp=a[i];  // em =4
	a[i]=a[i-1];  //  a1=5
	a[i-1]=temp;    // a0=4
	cout<<a[i-1]<<"  ";	
}
}
return 0;	}
